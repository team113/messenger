Future<T?> wrapped<T>(Future<T?> Function(ScopedDatabase) action) async {
  if (isClosed || db == null) {
    return null;
  }

  final Completer completer = Completer();
  _completers.add(completer);

  try {
    return await action(db!);
  } finally {
    completer.complete();
    _completers.remove(completer);
  }
}
