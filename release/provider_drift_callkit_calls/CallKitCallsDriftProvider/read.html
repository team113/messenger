Future<PreciseDateTime?> read(String id) async {
  return await safe<PreciseDateTime?>((db) async {
    final stmt = db.select(db.callKitCalls)..where((u) => u.id.equals(id));
    final CallKitCallRow? row = await stmt.getSingleOrNull();
    return row?.at;
  });
}
