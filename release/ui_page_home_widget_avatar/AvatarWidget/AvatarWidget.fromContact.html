factory AvatarWidget.fromContact(
  ChatContact? contact, {
  Key? key,
  Avatar? avatar,
  AvatarRadius? radius,
  double opacity = 1,
}) => AvatarWidget(
  key: key,
  avatar: avatar,
  title: contact?.name.val,
  color: (contact?.users.isEmpty ?? false)
      ? contact?.name.val.sum()
      : contact?.users.first.num.val.sum(),
  radius: radius,
  opacity: opacity,
);
