  if (mime == null) {
    if (path != null) {
      var type = lookupMimeType(path!);
      if (type != null) {
        mime = MediaType.parse(type);
      }
    } else if (bytes != null) {
      var type = lookupMimeType(name, headerBytes: bytes);
      if (type != null) {
        mime = MediaType.parse(type);
      }
    }
  }

  if (bytes != null) {
    _determineDimension();
  }
}
