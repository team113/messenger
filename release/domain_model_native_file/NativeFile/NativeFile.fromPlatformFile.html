factory NativeFile.fromPlatformFile(PlatformFile file) => NativeFile(
      name: file.name,
      size: file.size,
      path: PlatformUtils.isWeb ? null : file.path,
      bytes: file.bytes,
      stream: file.readStream?.asBroadcastStream(),
    );
