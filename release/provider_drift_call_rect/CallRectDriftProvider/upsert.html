Future<void> upsert(ChatId chatId, Rect rect) async {
  _cache[chatId] = rect;

  await safe((db) async {
    final Rect stored = _RectDb.fromDb(
      await db.into(db.callRectangles).insertReturning(
            rect.toDb(chatId),
            mode: InsertMode.insertOrReplace,
          ),
    );

    return stored;
  });
}
