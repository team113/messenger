int julianDayNumber() {
  final int c0 = ((month - 3) / 12).floor();
  final int x4 = year + c0;
  final int x3 = (x4 / 100).floor();
  final int x2 = x4 % 100;
  final int x1 = month - (12 * c0) - 3;
  return ((146097 * x3) / 4).floor() +
      ((36525 * x2) / 100).floor() +
      (((153 * x1) + 2) / 5).floor() +
      day +
      1721119;
}
