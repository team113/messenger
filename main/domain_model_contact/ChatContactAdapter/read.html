@override
ChatContact read(BinaryReader reader) {
  final numOfFields = reader.readByte();
  final fields = <int, dynamic>{
    for (int i = 0; i < numOfFields; i++) reader.readByte(): reader.read(),
  };
  return ChatContact(
    fields[0] as ChatContactId,
    name: fields[1] as UserName,
    users: (fields[2] as List?)?.cast<User>(),
    groups: (fields[3] as List?)?.cast<Chat>(),
    phones: (fields[5] as List?)?.cast<UserPhone>(),
    emails: (fields[4] as List?)?.cast<UserEmail>(),
    favoritePosition: fields[6] as ChatContactPosition?,
  );
}
