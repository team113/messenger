Future<void> leave(ChatId chatId, ChatCallDeviceId deviceId) async {
  Rx<OngoingCall>? call = _callsRepo[chatId];
  if (call != null) {
    call.value.state.value = OngoingCallState.ended;
    call.value.dispose();
    await _callsRepo.leave(chatId, deviceId);
  }
}
