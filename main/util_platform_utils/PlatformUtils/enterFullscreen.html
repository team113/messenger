static void enterFullscreen() {
  if (isWeb) {
    WebUtils.toggleFullscreen(true);
  } else if (isDesktop) {
    WindowManager.instance.setFullScreen(true);
  } else if (isMobile) {
    SystemChrome.setEnabledSystemUIMode(SystemUiMode.manual, overlays: []);
  }
}
