Widget handButton(CallController c, [double? scale]) => Obx(
      () => RoundFloatingButton(
        hint: c.isHandRaised.value

            : 'btn_call_hand_up'.l10n,
        onPressed: c.toggleHand,
        scale: scale ?? 1,
        children: [
          SvgLoader.asset(
            'assets/icons/hand_${c.isHandRaised.value ? 'down' : 'up'}.svg',
            width: 60,
          )
        ],
      ),
    );
