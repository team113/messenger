String? getSubtitle({User? partner}) {
  switch (kind) {
    case ChatKind.dialog:
      return partner?.getStatus();

    case ChatKind.group:
      return '${members.length} ${'label_subtitle_participants'.l10n}';

    case ChatKind.monolog:
    case ChatKind.artemisUnknown:
      return null;
  }
}
