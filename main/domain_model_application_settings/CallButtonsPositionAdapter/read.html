@override
CallButtonsPosition read(BinaryReader reader) {
  switch (reader.readByte()) {
    case 0:
      return CallButtonsPosition.appBar;
    case 1:
      return CallButtonsPosition.contextMenu;
    case 2:
      return CallButtonsPosition.top;
    case 3:
      return CallButtonsPosition.bottom;
    case 4:
      return CallButtonsPosition.more;
    default:
      return CallButtonsPosition.appBar;
  }
}
